33       *
34       * @param gitRepositoryUri
35       *            git repository URL.
36       * @param branchName
37       *            branch name usually "gh-pages"
38       * @param resource
39       *            resource
40       */
41      public GitUri(final String gitRepositoryUri, final String branchName,
42              final String resource) {
43          this.gitRepositoryUri = gitRepositoryUri;
44          this.branchName = branchName;

48      /**
49       *
50       * @param gitUri
51       *            valid git URI (i.e. no git: schema).
52       */
53      private GitUri(final URI gitUri) {
54          branchName = gitUri.getQuery();
55          final String asciiUriString = gitUri.toASCIIString();
56          gitRepositoryUri = asciiUriString.substring(0,
57                  asciiUriString.lastIndexOf('?'));
58          resource = gitUri.getFragment();
59      }

84      /**
85       * Resolves a new {@link GitUri} with the fragment. The schema portion will
86       * find the last occurrence of a URI path segment with a "?" character as
87       * the Git URI with the branch name.
88       *
89       * @param fragment
90       *            may contain escaped characters.
91       * @return resolved {@link GitUri}
92       */
93      public GitUri resolve(final String fragment) {
94          // TODO clean this up so it is less "hacky"
95          final String decodedFragment;
96          try {
97              decodedFragment = URLDecoder.decode(fragment, "UTF-8");
98          } catch (final UnsupportedEncodingException e) {
99              // should not happen.
100             throw new IllegalStateException(e);
101         }
102         final URI combined;
103         if (resource != null) {
104             combined = URI.create(gitRepositoryUri
105                     + resource.replace("##", "#"));
106         } else {
107             combined = URI.create(gitRepositoryUri);
108         }
109         final String slashAppendedUri = combined.toASCIIString();
110         final URI resolvedUri = URI.create(slashAppendedUri).resolve(
111                 decodedFragment.replace(" ", "%2520"));
112         final StringBuilder resolved = new StringBuilder(
113                 resolvedUri.toASCIIString());
114 
115         final int lastQuestionMark = resolved.lastIndexOf("?");
116         if (lastQuestionMark == -1) {
117             return new GitUri(URI.create(format("%s?%s#%s", gitRepositoryUri,
118                     branchName, decodedFragment.replace(" ", "%20"))));
119         }
120         final int lastRelevantSlash = resolved.indexOf("/", lastQuestionMark);
121         final int lastRelevantHash = resolved.indexOf("#", lastQuestionMark);
122 
123         if (lastRelevantSlash > 0 && lastRelevantHash >= lastRelevantSlash) {
124             resolved.insert(lastRelevantSlash, '#');
125         }
126 
127         return new GitUri(URI.create(resolved.toString()));
128     }
