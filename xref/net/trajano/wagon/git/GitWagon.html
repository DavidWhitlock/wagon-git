14  /**
15   * Git Wagon. Due to issues with the way maven-site-plugin is improperly sending
16   * requests that assume the target repository is a file system, the handling of
17   * git URIs fails. This performs an inefficient, but working method of creating
18   * a clone per request, but only once per Git repository.
19   */
20  @Component(role = Wagon.class, hint = "git", instantiationStrategy = "per-lookup")
21  public class GitWagon extends AbstractGitWagon {
22      /**
23       * Constructs the object from a URI string that contains "git:" schema.
24       * {@inheritDoc}
25       */
26      @Override
27      public GitUri buildGitUri(final String repositoryUrl)
28              throws ConnectionException, AuthenticationException {
29          try {
30              final URI uri = new URI(repositoryUrl.replace("##", "#"));
31              final URI gitUri = new URI(uri.getSchemeSpecificPart());
32              final String branchName = gitUri.getQuery();
33              final String asciiUriString = gitUri.toASCIIString();
34              final String gitRepositoryUri = asciiUriString.substring(0,
35                      asciiUriString.indexOf('?'));
36              final String resource = uri.getFragment();
37              return new GitUri(gitRepositoryUri, branchName, resource);
38          } catch (final URISyntaxException e) {
39              throw new ConnectionException(e.getMessage());
40          }
41      }
42  }
