35      /**
36       * Builds a GitUri from a GitHub Pages URL.  It performs a DNS lookup for
37       * the CNAME if the host does not match {@link #GITHUB_PAGES_HOST_PATTERN}.
38       * 
39       * @param gitHubPagesUrl
40       *            GitHub Pages URL
41       * @return a GitURI based on the GitHubPages URL
42       * @throws TextParseException
43       */
44      public static GitUri buildFromGithubPages(final String gitHubPagesUrl)
45              throws TextParseException {
46          final URI uri = URI.create(
47                  URI.create(gitHubPagesUrl).getSchemeSpecificPart()).normalize();
48          final Matcher m = GITHUB_PAGES_HOST_PATTERN.matcher(uri.getHost());
49          final String username;
50          if (m.matches()) {
51              username = m.group(1);
52          } else {
53              final String cnameHost = getCnameForHost(uri.getHost());
54              final Matcher m2 = GITHUB_PAGES_HOST_PATTERN.matcher(cnameHost);
55              if (!m2.matches()) {
56                  throw new RuntimeException("Invalid host for github pages "
57                          + gitHubPagesUrl);
58              }
59              username = m2.group(1);
60          }
61          final Matcher pathMatcher = GITHUB_PAGES_PATH_PATTERN.matcher(uri
62                  .getPath());
63          pathMatcher.matches();
64          return new GitUri("ssh://git@github.com/" + username + "/"
65                  + pathMatcher.group(1) + ".git", "gh-pages", "");
66      }
67  
68      private static String getCnameForHost(final String host)
69              throws TextParseException {
70          final Lookup lookup = new Lookup(host, Type.CNAME);
71          lookup.run();
72          if (lookup.getAnswers().length == 0) {
73              return null;
74          }
75          return ((CNAMERecord) lookup.getAnswers()[0]).getTarget().toString();
76      }

93      /**
94       * Constructs the object from a URI string that contains "git:" schema.
95       *
96       * @param uriString
97       *            URI string.
98       * @throws URISyntaxException
99       *             parsing exception
100      */
101     public GitUri(final String uriString) throws URISyntaxException {
102         final URI uri = new URI(uriString.replace("##", "#"));
103         final URI gitUri = new URI(uri.getSchemeSpecificPart());
104         branchName = gitUri.getQuery();
105         final String asciiUriString = gitUri.toASCIIString();
106         gitRepositoryUri = asciiUriString.substring(0,
107                 asciiUriString.indexOf('?'));
108         resource = uri.getFragment();
109     }

163     /**
164      * Resolves a new {@link GitUri} with the fragment. The schema portion will
165      * find the last occurrence of a URI path segment with a "?" character as
166      * the Git URI with the branch name.
167      *
168      * @param fragment
169      *            may contain escaped characters.
170      * @return resolved {@link GitUri}
171      */
172     public GitUri resolve(final String fragment) {
173         // TODO clean this up so it is less "hacky"
174         final String decodedFragment;
175         try {
176             decodedFragment = URLDecoder.decode(fragment, "UTF-8");
177         } catch (final UnsupportedEncodingException e) {
178             // should not happen.
179             throw new IllegalStateException(e);
180         }
181         final URI combined;
182         if (resource != null) {
183             combined = URI.create(gitRepositoryUri
184                     + resource.replace("##", "#"));
185         } else {
186             combined = URI.create(gitRepositoryUri);
187         }
188         final String slashAppendedUri = combined.toASCIIString();
189         final URI resolvedUri = URI.create(slashAppendedUri).resolve(
190                 decodedFragment.replace(" ", "%2520"));
191         final StringBuilder resolved = new StringBuilder(
192                 resolvedUri.toASCIIString());
193 
194         final int lastQuestionMark = resolved.lastIndexOf("?");
195         if (lastQuestionMark == -1) {
196             return new GitUri(URI.create(format("%s?%s#%s", gitRepositoryUri,
197                     branchName, decodedFragment.replace(" ", "%20"))));
198         }
199         final int lastRelevantSlash = resolved.indexOf("/", lastQuestionMark);
200         final int lastRelevantHash = resolved.indexOf("#", lastQuestionMark);
201 
202         if (lastRelevantSlash > 0 && lastRelevantHash >= lastRelevantSlash) {
203             resolved.insert(lastRelevantSlash, '#');
204         }
205 
206         return new GitUri(URI.create(resolved.toString()));
207     }
